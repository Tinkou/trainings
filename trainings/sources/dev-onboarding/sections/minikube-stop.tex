\begin{frame}[fragile]
	\frametitle{Using minikube}
	
	For the following, make sure that minikube is stopped:
	\begin{block}{Command terminal}
		\begin{verbatim}
			minikube stop
		\end{verbatim}
	\end{block}
	
	Check minikube status:
	\begin{block}{Command terminal}
		\begin{verbatim}
			minikube status
		\end{verbatim}
	\end{block}
\end{frame}
